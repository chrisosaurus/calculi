\textbf{Expressions}

\begin{grammar}

    <e> ::=  x
        \alt $\lambda$ x.<e>
        \alt $<e>_1$ $<e>_2$
        \alt unit
        \alt L
        \alt new <e>
        \alt read <e>
        \alt write $<e>_1$ $<e>_2$
        \alt true
        \alt false
        \alt if $<e>_1$ $<e>_2$ $<e>_3$

\end{grammar}

\textbf{Trivial Expressions}

\begin{grammar}

    <v> ::=  L
        \alt $\lambda$ x.<e>
        \alt unit
        \alt true
        \alt false

\end{grammar}

\textbf{Evaluation Contexts}

\begin{grammar}

    <C[\textbullet]> ::=  \textbullet
                     \alt <C[\textbullet]> $<e>_2$
                     \alt $<v>_1$ <C[\textbullet]>
                     \alt new <C[\textbullet]>
                     \alt read <C[\textbullet]>
                     \alt write <C[\textbullet]> $<e>_2$
                     \alt write $<e>_1$ <C[\textbullet]>
                     \alt if <C[\textbullet]> $<e>_2$ $<e>_3$

\end{grammar}

\textbf{Operational semantics}

\begin{center}
\framebox{
                $
                <e, \sigma>
                \quad
                \rightarrow
                \quad
                <e', \sigma'>
                $
}
\end{center}

\begin{prooftree}
    \AxiomC{ $
                <e, \sigma>
                \quad
                \rightarrow
                \quad
                <e', \sigma'>
             $ }
    \RightLabel{\it context}
    \UnaryInfC{ $
                <C[e], \sigma>
                \quad
                \rightarrow
                \quad
                <C[e'], \sigma'>
                $ }
\end{prooftree}

\begin{prooftree}
    \AxiomC{ $
             $ }
    \RightLabel{\it application}
    \UnaryInfC{ $
                <(\lambda x . e_1) v_2, \sigma>
                \quad
                \rightarrow
                \quad
                <e_1 [v_2 / x], \sigma>
                $ }
\end{prooftree}

\begin{prooftree}
    \AxiomC{ $
                L \ fresh
             $ }
    \RightLabel{\it new}
    \UnaryInfC{ $
                <new \ v, \sigma>
                \quad
                \rightarrow
                \quad
                <L, \sigma[L \mapsto v]>
                $ }
\end{prooftree}

\begin{prooftree}
    \AxiomC{ $
             $ }
    \RightLabel{\it read}
    \UnaryInfC{ $
                <L, \sigma[L \mapsto v]>
                \quad
                \rightarrow
                \quad
                <v, \sigma>
                $ }
\end{prooftree}

\begin{prooftree}
    \AxiomC{ $
             $ }
    \RightLabel{\it write}
    \UnaryInfC{ $
                <write \ L \ v_2, \sigma[L \mapsto v_1]>
                \quad
                \rightarrow
                \quad
                <unit, \sigma[L \mapsto v_2]>
                $ }
\end{prooftree}

\begin{prooftree}
    \AxiomC{ $
                <e_1, \sigma>
                \quad
                \rightarrow
                \quad
                <true, \sigma'>
             $ }
    \RightLabel{\it if-true}
    \UnaryInfC{ $
                <if \ e_1 \ e_2 \ e_3, \sigma>
                \quad
                \rightarrow
                \quad
                <e_2, \sigma'>
                $ }
\end{prooftree}

\begin{prooftree}
    \AxiomC{ $
                <e_1, \sigma>
                \quad
                \rightarrow
                \quad
                <false, \sigma'>
             $ }
    \RightLabel{\it if-false}
    \UnaryInfC{ $
                <if \ e_1 \ e_2 \ e_3, \sigma>
                \quad
                \rightarrow
                \quad
                <e_3, \sigma'>
                $ }
\end{prooftree}
